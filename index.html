__main

COPY

OUTER

INNER

Factorial (Subroutine)

AREA factorial,CODE,READONLY
EXPORT __main

__main PROC
LDR r0,=0x20000000
LDR r1,[r0,#0]
BL fact
STR r3,[r0,#4]

l b l
ENDP

fact PROC
mov r2,r1
mov r3,#1

loop
mul r3,r2,r3
sub r2,r2,#1
cmp r2,#0
bne loop

BX LR
ENDP
end

Recursive Factorial

AREA factorialrec,CODE,READONLY
EXPORT __main

__main PROC
LDR r0,=0x20000000
LDR r1,[r0,#0]
BL fact
STR r1,[r0,#4]

l b l
ENDP

fact PROC
sub sp,sp,#8
STR LR,[sp,#4]
STR r1,[sp,#0]

cmp r1,#1
bgt L1
b lpop

L1
sub r1,r1,#1
BL fact

lpop
mov r12,r1
LDR r1,[sp,#0]
LDR LR,[sp,#4]
ADD sp,sp,#8

mul r1,r12,r1
mov pc,LR

ENDP
END

STM32 Timer Delay

#include

void delay(){

RCC->APB1ENR |=(1<<0);

TIM2->PSC=7199;
TIM2->ARR=499;
TIM2->SR=0;

TIM2->CR1=1;

while((TIM2->SR & 1)==0);

TIM2->CR1=0;

}

int main(){

RCC->APB2ENR |=0xFC;
